Examples

Simple panel


    <link href="coreui-panel.css" rel="stylesheet"/>
    <script src="coreui-panel.js"></script>

    <div id="example-simple-content"></div>

    <script>
        document.addEventListener('DOMContentLoaded', function () {

            var panel = new CoreUIPanel({
                "type":  "panel",
                "name":  "simple",
                "title": "Component Panel",
                "subtitle": "CoreUI Framework",
                "actions": "<span class=\"help\" title=\"help\">?<\/span>",
                "content": ["Your content"]
            });

            panel.render('example-simple-content');
        });
    </script>
                            

    $panel = new \CoreUI\Panel('simple');

    $panel->setTitle('Component Panel', 'CoreUI Framework');\
    $panel->setActions('<span class="help" title="help">?</span>');
    $panel->addContent( 'Your content' );

    echo json_encoded($panel->render()); // JSON output


Togglable tabs


    <link href="coreui-panel.css" rel="stylesheet"/>
    <script src="coreui-panel.js"></script>

    <div id="example-tabs-content"></div>

    <script>
        document.addEventListener('DOMContentLoaded', function () {

            var panel = new CoreUIPanel({
                "type": "panel",
                "name": "tabs",
                "tabs": [
                    {"id": "1", "title": "Home", "type": "tab", "active": true},
                    {"id": "2", "title": "Profile", "type": "tab", "active": false},
                    {"id": "3", "title": "Disabled", "type": "tab", "disabled": true},
                    {
                        "title": "Dropdown",
                        "type": "combotab",
                        "elements": [
                            {"id": "4", "title": "Tab title 4", "type": "element", "disabled": true},
                            {"id": "5", "title": "Tab title 5", "type": "element", "active": false},
                            {"type": "divider"},
                            {"id": "6","title": "Tab title 6","type": "element","active": false}
                        ]
                    }
                ],
                "content": [ "Tab 1" ]
            });

            panel.render('example-tabs-content');

            panel.on('click-tab.coreui.panel', function (event, name, tab_id) {
                $('#coreui-panel-' + name + ' .coreui-panel-body')
                    .html('Tab ' + tab_id);
            });
        });
    </script>
                            

    $panel = new \CoreUI\Panel('tabs');

    $panel->addTab('Home',     '1');
    $panel->addTab('Profile',  '2');
    $panel->addTab('Disabled', '3', '', true);
    $panel->addComboTab('Dropdown')
        ->addItem('Tab title 3', '4', '', true)
        ->addItem('Tab title 4', '5')
        ->addBreak()
        ->addItem('Tab title 5', '');

    $panel->addContent( 'Tab 1' );

    echo json_encoded($panel->render()); // JSON output


Insert components


    <link href="coreui-panel.css" rel="stylesheet"/>
    <script src="coreui-panel.js"></script>

    <div id="example-components-content"></div>

    <script>
        document.addEventListener('DOMContentLoaded', function () {

            var panel = new CoreUIPanel({
                "type": "panel",
                "name": "components",
                "title": "Component Panel",
                "subtitle": "CoreUI Framework",
                "content": [
                    "Your content 1 <br><br>",
                    {
                        "type": "html",
                        "content" : "Your content 2 <br><br>"
                    },
                    {
                        "type":  "panel", // and other components
                        "name":  "sub-panel",
                        "title": "Sub Panel",
                        "content": [ "Your content 3" ]
                    }
                ]
            });
            panel.render('example-components-content');
        });
    </script>

                            

    // Panel 1
    $panel = new \CoreUI\Panel('components');
    $panel->setTitle('Component Panel', 'CoreUI Framework');
    $panel->addContent( 'Your content 1 <br><br>' );
    $panel->addContent( 'Your content 2 <br><br>' );

    // Panel 2
    $sub_panel = new \CoreUI\Panel('sub-panel');
    $sub_panel->setTitle('Sub Panel');
    $sub_panel->addContent( 'Your content 3' );

    $panel->addContent($sub_panel->render());

    echo json_encoded($panel->render()); // JSON output


Ajax


    <link href="coreui-panel.css" rel="stylesheet"/>
    <script src="coreui-panel.js"></script>

    <div id="example-ajax-content"></div>

    <script>
        document.addEventListener('DOMContentLoaded', function () {

             var panel = new CoreUIPanel({
                "type": "panel",
                "name": "ajax",
                "title": "Component Panel",
                "actions": "<span class=\"help\" title=\"help\">?<\/span>",
                "subtitle": "CoreUI Framework",
                "tabs": [
                    {"id":"tab1", "title":"Home",     "type":"tab", "active":true,  "url":"data/tab1.txt"},
                    {"id":"tab2", "title":"Profile",  "type":"tab", "active":false, "url":"data/tab2.json"},
                    {"id":"tab3", "title":"Disabled", "type":"tab", "disabled":true},
                    {
                        "title": "Dropdown",
                        "type":  "combotab",
                        "elements": [
                            {"id":"tab4", "title":"Tab title 3", "type":"element", "disabled":true},
                            {"id":"tab5", "title":"Tab title 4", "type":"element", "active":false, "url":"data/tab3.json"},
                            {"type": "divider"},
                            {"id":"tab6", "title":"Tab title 5", "type":"element", "active":false, "url":"data/tab4.json"}
                        ]
                    }
                ],
                "content": ["Your content 1"]
            });
            panel.render('example-ajax-content');
            panel.on('click-tab.coreui.panel', function (event, name, tab_id) {

                $.ajax({
                    url: this.href,
                    method: 'get',
                    success: function (result) {
                        var content = CoreUIPanel.renderContent(result);
                        $('#coreui-panel-' + name + ' .coreui-panel-body').html(content);
                    }
                });

                event.preventDefault();
            });
        });
    </script>
                            

    $panel = new \CoreUI\Panel('ajax');
    $panel->setTitle('Component Panel', 'CoreUI Framework');

    $panel->setActions('<span class="help" title="help">?</span>');
    $panel->addTab('Home',     'tab1', 'data/tab1.json');
    $panel->addTab('Profile',  'tab2', 'data/tab2.json');
    $panel->addTab('Disabled', 'tab3', '', true);
    $panel->addComboTab('Dropdown')
        ->addItem('Tab title 3', 'tab4', '', true)
        ->addItem('Tab title 4', 'tab5', 'data/tab3.json')
        ->addBreak()
        ->addItem('Tab title 5', 'tab6', 'data/tab4.json');

    switch ($panel->getActiveTab()) {
        case 'tab1': $panel->addContent( 'Your content 1' ); break;
        case 'tab2': $panel->addContent( 'Your content 2' ); break;
        case 'tab3': $panel->addContent( 'Your content 3' ); break;
        case 'tab4': $panel->addContent( 'Your content 4' ); break;
        case 'tab5': $panel->addContent( 'Your content 5' ); break;
        case 'tab6': $panel->addContent( 'Your content 6' ); break;
    }

    echo json_encoded($panel->render()); // JSON output

Options

Name type default description
type string The data type. The panel must be specified for correct interpretation.
name string Unique panel name
view string default Identify the appearance of the panel. Can have different values: default, primary, success, warning, danger. But there may be other values if you have you will create your personal styles.
title string The title of the panel.
subtitle string Text of additional panel title.
actions string Any html text to create additional panel controls.
tabs array Array of objects containing properties for creating bookmarks.
content array An array with the contents of the panel. Elements can be strings or objects.

Methods

CoreUIPanel.get('panel-name')

Receiving a previously created panel by its name.

panel.render('element-id')

Forms and inserts the generated html panel into the specified element. The element's parameter can contain the page element identifier or the page element's object. If there is no specified parameter, the method will return the html panel.

CoreUIPanel.renderContent({ ... })

Forms from the passed parameter html returns it.

CoreUIPanel.initEvents(element)

Initialize associated events for the panel. The parameter can contain the dom element in which you need to look for panels to initialize events. When using the method panel.render('element-id') event initialization is performed automatically.

CoreUIPanel.onGlobal('click-tab.coreui.panel', function() { ... })

Set an event that can occur in any of the panels created.

panel.on('click-tab.coreui.panel', function() { ... })

Set an event that can occur in the current panel.

panel.activeTab('tab-id', context, event)

Set the activity in the specified tab. The first parameter is the tab ID, context and event are the context and the event parameter with which the events specified for this action should be called.

Events

Event Type Description
click-tab.coreui.panel This event occurs when the user clicks on a bookmark or when the method is called panel.activeTab('tab-id', callback).
show.coreui.panel This event occurs before the panel is inserted into the page using the method panel.render('element-id').
shown.coreui.panel This event occurs after the panel is pushed onto the page using the method panel.render('element-id').
